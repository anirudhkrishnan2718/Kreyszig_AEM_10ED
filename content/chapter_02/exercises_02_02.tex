          \begin{align}
              4y'' - 25y               & = 0                                \\
              a                        & = 0                              &
              b                        & = \frac{-25}{4}                    \\
              a^{2} - 4b               & = 25 > 0                           \\
              \lambda_{1}, \lambda_{2} & = \frac{0 \pm \sqrt{25}}{2}      &
                                       & = \pm \frac{5}{2}                  \\
              y                        & = c_{1}e^{5x/2} + c_{2}e^{-5x/2}
          \end{align}
          Checking by substitution,
          \begin{align}
              y'  & = \frac{5c_{1}}{2}e^{5x/2} - \frac{5c_{2}}{2}e^{-5x/2}   \\
              y'' & = \frac{25c_{1}}{4}e^{5x/2} + \frac{25c_{2}}{4}e^{-5x/2} \\
              y'' & = \frac{25}{4}y
          \end{align}
