    \item Using the sum of normal RVs,

              \mu                        & = 100 + 5 = 105                           &

              P(104 < Y < 106)           & = P\left( \frac{-2}{\sqrt{5}} < Z <
              \frac{2}{\sqrt{5}} \right) &

          \end{align}

    \item Using the sum of normal RVs, $Y = nX$,
          \begin{align}
              \mu         & = 40n                                      &
              \sigma      & = \sqrt{4n}                                  \\
              P(Y > 2000) & = P\left( Z > \frac{2000 - 40n}{2\sqrt{n}}
              \right)     &
                          & = 0.05                                       \\
              \frac{1000}{\sqrt{n}} - 20\sqrt{n}
                          & = 1.64485                                  &
              n           & = (7.03)^2 = 49.421                          \\
